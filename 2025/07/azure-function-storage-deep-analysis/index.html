// з®ҖеҢ–зҡ„Host IDз”ҹжҲҗз®—жі•
string GenerateHostId(string functionAppName)
{
    // 1. еҸ–Function AppеҗҚз§°
    // 2. иҪ¬жҚўдёәе°ҸеҶҷ
    // 3. жҲӘеҸ–еүҚ32дёӘеӯ—з¬Ұ
    // 4. з§»йҷӨзү№ж®Ҡеӯ—з¬ҰпјҲд»…дҝқз•ҷеӯ—жҜҚгҖҒж•°еӯ—гҖҒиҝһеӯ—з¬Ұпјү
    
    var hostId = functionAppName.ToLowerInvariant();
    if (hostId.Length > 32)
    {
        hostId = hostId.Substring(0, 32);
    }
    
    // жё…зҗҶж— ж•Ҳеӯ—з¬Ұе’Ңиҫ№з•Ңжғ…еҶө
    hostId = Regex.Replace(hostId, "[^a-z0-9-]", "-");
    hostId = hostId.Trim('-');
    
    return hostId;
}

жҖ§иғҪзӣ‘жҺ§дёҺе®№йҮҸи§„еҲ’

Storage Accountе…ій”®жҢҮж Ү

дәӢеҠЎйҮҸзӣ‘жҺ§пјҡ

жҢҮж ҮпјҡTotal requests/minute
йҳҲеҖјпјҡ
- еҚ•Function App: < 10,000 requests/min
- е…ұдә«Storage: < 50,000 requests/min
- е‘ҠиӯҰйҳҲеҖј: > 80% capacity

е»¶иҝҹзӣ‘жҺ§пјҡ

жҢҮж ҮпјҡAverage E2E Latency
йҳҲеҖјпјҡ
- Blobж“ҚдҪң: < 100ms
- Queueж“ҚдҪң: < 50ms  
- Tableж“ҚдҪң: < 30ms

е®№йҮҸи§„еҲ’е…¬ејҸпјҡ

йў„дј°StorageйңҖжұӮ =
